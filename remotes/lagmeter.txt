;display the lag


;create the alias
alias pingme scon -at1 .ctcpreply $!me LAG $ticks


;capture the ctcp reply
on *:CTCPREPLY:LAG*: {
  if ( $nick != $me ) halt
  echo -at * Lag: Lag for $server $+ : $calc(($ticks - $2) / 1000) $+ s
  halt
} 

;adds a menu popup
menu status,channel,query {
  $iif($server == $null,$style(2)) - Lag Meter:/pingme
}
